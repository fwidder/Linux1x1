~		-	Eigenes Homeverzeichnis
/bin		-	Hier liegen (fast) alle Terminal Programme
/dev		-	Hier befindet sich jegliche an den Rechner angeschlossene Hardware
/dev/null   	-	Das "nichts"
/dev/random 	-	Endlose viele Zufallszahlen
/dev/zero   	-	Endlos viele Nullen 
/etc		-	Hier liegen globale Konfigurationsdatein
/home		-	Hier liegen alle Homeverzeichnisse (Außer das von root)
/opt		-	Zusätzliche Software welche nicht via Paketmanager installiert wird
/root		-	Homeverzeichnis von root
/sbin		-	Programme welche nur von root ausgeführt werden können / dürfen
/srv		-	Hier liegen normalerweise Dateien für Web- oder FTP-Server
/tmp		-	Temporäres Verzeichnis (wird beim Neustart gelöscht)
/usr		-	Benutzer Software
/var		-	Variable Daten z.B.: Logs
